#include <stdlib.h>

struct Libro {
    char titulo[50];
    char autor[50];
    int precio;
    int likes;
};

int libros_con_likes(struct Libro libros[], int num_libros, int N, char vector_resultado[]) {
int i, contador = 0;
char temp[100];
for (i = 0; i < num_libros; i++) {
	if (libros[i].likes > N) {
            sprintf(temp, "%s*%d/", libros[i].titulo, libros[i].precio);
            strcat(vector_resultado, temp);
            contador++;
        }
    }

    return contador; 
}
